# Eval UI Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** A workbench to run a route, draw a candidate "better" path, and understand — with per-segment cost attribution and a calibration loop — why the engine didn't choose it, with saved cases forming a manually-labeled regression suite.

**Architecture:** One path *scorer* walks any polyline across the DEM grid and sums the exact `calculate_move_cost()` A\* minimizes; both the engine's optimal path and the user's drawn path are scored identically under the same `RouteOptions`, so the comparison is honest by construction. A frozen `ScoredPath` / `EvalCase` API contract lets frontend and backend tracks build in parallel against one interface.

**Tech Stack:** Backend FastAPI + numpy (`engine_v2`), Python venv `trail_env`. Frontend React 19 + Vite + TypeScript + react-leaflet 5. No new runtime dependencies.

```ts
// frontend/src/services/evalApi.ts
import { API_BASE } from './api'
import type { Coordinate, RouteOptions } from './api'

export interface ScoredSegment { from: Coordinate; to: Coordinate; cost: number; factors: Record<string, number>; dominantFactor: string }
export interface ScoredPath { path: Coordinate[]; snapped: boolean; totalCost: number; distanceM: number; elevationGainM: number; segments: ScoredSegment[] }
export interface EvalLabel { ts: string; verdict: 'pass' | 'fail' | 'unsure'; note?: string }
export interface EvalCase { id: string; name: string; notes?: string; start: Coordinate; end: Coordinate; options: RouteOptions; referencePath: Coordinate[]; labels: EvalLabel[] }

export async function scorePath(path: Coordinate[], options: RouteOptions, snap: 'none' | 'trail' = 'none'): Promise<ScoredPath> {
  const r = await fetch(`${API_BASE}/api/eval/score-path`, {
    method: 'POST', headers: { 'Content-Type': 'application/json' },
    body: JSON.stringify({ path, options, snap }),
  })
  if (!r.ok) throw new Error(`score-path ${r.status}`)
  return r.json()
}
export async function listCases(): Promise<EvalCase[]> { const r = await fetch(`${API_BASE}/api/eval/cases`); if (!r.ok) throw new Error(`${r.status}`); return r.json() }
export async function saveCase(c: EvalCase): Promise<EvalCase> { const r = await fetch(`${API_BASE}/api/eval/cases`, { method: 'POST', headers: { 'Content-Type': 'application/json' }, body: JSON.stringify(c) }); if (!r.ok) throw new Error(`${r.status}`); return r.json() }
export async function deleteCase(id: string): Promise<void> { const r = await fetch(`${API_BASE}/api/eval/cases/${id}`, { method: 'DELETE' }); if (!r.ok) throw new Error(`${r.status}`) }
```

```python
# backend/app/engine_v2/scoring.py
from typing import List, Tuple


def rasterize_segment(r0: int, c0: int, r1: int, c1: int) -> List[Tuple[int, int]]:
    """Supercover line: all grid cells the segment passes through, contiguous
    with 8-connected steps. Endpoints inclusive."""
    cells = [(r0, c0)]
    dr, dc = abs(r1 - r0), abs(c1 - c0)
    sr = 1 if r1 > r0 else -1
    sc = 1 if c1 > c0 else -1
    r, c = r0, c0
    err = dr - dc
    while (r, c) != (r1, c1):
        e2 = 2 * err
        if e2 > -dc:
            err -= dc; r += sr
        if e2 < dr:
            err += dr; c += sc
        cells.append((r, c))
    return cells


def score_polyline_cells(pf, cells: List[Tuple[int, int]], straight_line_distance: float):
    """Walk contiguous grid cells, summing pf.calculate_move_cost breakdowns."""
    total = 0.0
    factors = {"base": 0.0, "terrain": 0.0, "slope": 0.0, "sustained": 0.0, "deviation": 0.0}
    steep = 0.0
    dist = 0.0
    for (r0, c0), (r1, c1) in zip(cells, cells[1:]):
        bd = pf.calculate_move_cost(r0, c0, r1, c1, straight_line_distance, dist, steep, return_breakdown=True)
        total += bd["cost"]
        for k, v in bd["factors"].items():
            factors[k] += v
        steep = bd["new_steep_distance"]
        dist += pf.resolution * ((r1 - r0) ** 2 + (c1 - c0) ** 2) ** 0.5
    return {"total": total, "factors": factors, "distance": dist}
```

- [ ] **Step 7: Implement `TrailFinderServiceV2.score_path`**

Mirror `_find_route_sync`'s load block (bounding box over the *polyline* extent via `calculate_bounding_box` on min/max corner Coordinates; load elevation; build terrain grid; `_make_pathfinder`). Then convert each polyline vertex with `pf.lat_lon_to_grid`, `rasterize_segment` per edge, `score_polyline_cells`, and build `ScoredPath`. Elevation gain = sum of positive elevation deltas along cells. Per-segment: score each edge's cells separately to fill `ScoredSegment.factors` and `dominantFactor = max(factors excluding "base")`.

```python
# sketch — real code in service.py
def score_path(self, path, options):
    options = options or {}
    corners = _bbox_coords(path)  # Coordinate(min lat,min lon), Coordinate(max,max)
    bounds = self.calculate_bounding_box(corners[0], corners[1], options.get("buffer"))
    with self._load_lock:
        self.elevation_lib.load_area(bounds)
        elevation, meta = self.elevation_lib.get_elevation_array(bounds)
        transform = _affine_from_bounds(bounds, elevation.shape)
        terrain_grid = self.path_layer.get_grid(bounds, elevation.shape, transform)
    pf = self._make_pathfinder(elevation, transform, terrain_grid, options)
    verts = [pf.lat_lon_to_grid(p.lat, p.lon) for p in path]
    sld = _grid_straight_line(verts[0], verts[-1], pf.resolution)
    segments, total, dist, egain = [], 0.0, 0.0, 0.0
    running_steep = 0.0
    for (a, b), (pa, pb) in zip(zip(verts, verts[1:]), zip(path, path[1:])):
        cells = rasterize_segment(*a, *b)
        s = score_polyline_cells(pf, cells, sld)  # extend to accept & return running_steep, egain
        segments.append(ScoredSegment(**{"from": pa, "to": pb, "cost": s["total"],
                                         "factors": s["factors"],
                                         "dominantFactor": _dominant(s["factors"])}))
        total += s["total"]; dist += s["distance"]
    return ScoredPath(path=path, snapped=False, totalCost=total, distanceM=dist,
                      elevationGainM=egain, segments=segments)
```
